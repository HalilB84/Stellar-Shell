 // Floating particles
            Repeater {
                model: 1000
                Rectangle {
                    required property int modelData
                    
                    width: 2
                    height: 2
                    radius: 1
                    color: Colors.cluGlow
                    opacity: 0.7
                    
                    property real baseX: Math.random() * (bar.width - width)
                    x: baseX
                    y: bar.height
                    
                    SequentialAnimation on y {
                        loops: Animation.Infinite
                        NumberAnimation {
                            to: -height
                            duration: 4000 + (modelData * 500) 
                            easing.type: Easing.InOutSine
                        }
                        ScriptAction { 
                            script: {
                                baseX = Math.random() * (bar.width - width)
                            }
                        }
                        NumberAnimation {
                            to: bar.height
                            duration: 0 
                        }
                        PauseAnimation { duration: modelData * 300 }
                    }
                    
                    //horizontal drift 
                    SequentialAnimation on x {
                        loops: Animation.Infinite
                        NumberAnimation {
                            to: baseX + 20
                            duration: 2000 + (modelData * 200)
                            easing.type: Easing.InOutSine
                        }
                        NumberAnimation {
                            to: baseX - 20
                            duration: 2000 + (modelData * 200)
                            easing.type: Easing.InOutSine
                        }
                    }
                }
            }
